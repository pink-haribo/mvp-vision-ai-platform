def is_yolo(path):
    # êµ¬ì¡°: images/*.jpg + labels/*.txt
    has_images = (path / 'images').exists()
    has_labels = (path / 'labels').exists()

    if has_images and has_labels:
        # .txt íŒŒì¼ í˜•ì‹ ê²€ì¦
